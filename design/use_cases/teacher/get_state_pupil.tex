\begin{uc}{Get state of pupil}

    \begin{uc-mss}
    \item The teacher is taken to a page that gives an overview of all pupils joining a competition with their stats.
    \item The teacher clicks on "Close stats".
    \item The teacher is redirected back to his landing-page.
    \end{uc-mss}

    \begin{uc-pre}
    \item The teacher is logged in.
    \item The teacher opened the organizer's competition page.
    \item The teacher clicked on "View competitions".
    \item The teacher clicked on a certain competition.
    \end{uc-pre}

    \begin{uc-post}
    \item Stats page has been closed
    \end{uc-post}

    \begin{uc-trig}
        The teacher clicks on "View stats" next to the competition in the competition's list.
    \end{uc-trig}

\end{uc}
