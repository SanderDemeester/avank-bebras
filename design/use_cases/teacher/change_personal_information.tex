\begin{uc}{Change personal information}

    \begin{uc-mss}
    \item The teacher clicks on "Edit personal information".
    \item The teacher is taken to a page that allows him to change his personal information.
    \item The teacher changes his personal information.
    \item The teacher clicks on "Submit".
    \item The teacher is redirected back to his landing-page.
    \end{uc-mss}

    \begin{uc-ext}

        \begin{uc-fail}{4}{Repeated input of wrong information}
        \item Error message: Repeated input of invalid information.
            \return{2}
        \end{uc-fail}

        \begin{uc-fail}{4}{Empty fields}
        \item Error message: There are emtpy fields.
        \item Identify the empty fields.
            \return{2}
        \end{uc-fail}

    \end{uc-ext}

    \begin{uc-pre}
    \item The teacher is logged into the system.
    \item The teacher is on his personal web page.
    \end{uc-pre}

    \begin{uc-post}
    \item The teacher's personal information has been changed.
    \end{uc-post}

    \begin{uc-trig}
        The user clicked on "Change personal information" on his personal web page.
    \end{uc-trig}

\end{uc}
