
\begin{uc}{Edit Question-set}

    \extends{Database errors}

    \begin{uc-mss}
    \item Change the following fields:
        \begin{itemize}
            \item Name
            \item Active (checkbox)
            \item Competition (not editable on this page)
        \end{itemize}
    \item Edit Question-set list \extend{Edit Question-set list}
    \item Click on the Save button.
    \end{uc-mss}

    \begin{uc-ext}

        \begin{uc-fail}{3}{Active selected AND non-activated Questions in Question-set list}
        \item Active box is unchecked.
        \item Warning message: There are non-active questions in the list, this set can't
        be activated until all of these questions are marked as active.
        \return{1}
        \end{uc-fail}

        \begin{uc-fail}{3}{Active selected AND empty Question-set list}
        \item Active box is unchecked.
        \item Warning message: A question-set cannot be activated without any questions.
        \return{1}
        \end{uc-fail}

    \end{uc-ext}

    \begin{uc-pre}
    \item User is authenticated as Organizer
    \item Current page: Manage Question-set
    \end{uc-pre}

    \begin{uc-post}
    \item Question-set is updated.
    \item Redirect to page: Manage Question-set
    \end{uc-post}

    \begin{uc-trig}
        Organizer click on the Edit button in the row of a Question-set inside the
        table.
    \end{uc-trig}

\end{uc}
