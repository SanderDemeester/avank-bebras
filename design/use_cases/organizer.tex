\begin{subsubsection}{Use Case Edit Pupil}
\vspace{8 pt}

\textbf{Main Success Scenario}
\begin{enumerate*}
\item Change text in the following fields: Name, Email Address, ClassGroup, Gender, Birthdate.
\item Click on the Save button.
\end{enumerate*}

\vspace{4 pt}

\textbf{Extensions}


\textbf{$<$MSS STEP 2$>$  [ClassGroup ID doesn't exist]}
\begin{enumerate*}
\itemdot Error message: The specified ClassGroup ID does not exist, please fill in an
existing one.
\itemdot Return to MSS step 1.
\end{enumerate*}

\textbf{$<$MSS STEP 2$>$  [Invalid Email Address]}
\begin{enumerate*}
\itemdot Error message: Please enter a valid email address.
existing one.
\itemdot Return to MSS step 1.
\end{enumerate*}

\textbf{$<$MSS STEP 2$>$  [Email Address already available in database of Pupils.]}
\begin{enumerate*}
\itemdot Error message: This email address is already assigned to a pupil.
\itemdot Return to MSS step 1.
\end{enumerate*}

\textbf{$<$MSS STEP 2$>$  [Not all fields are filled in]}
\begin{enumerate*}
\itemdot Error message: Please make sure all the fields are filled in.
\itemdot Return to MSS step 1.
\end{enumerate*}

\vspace{4 pt}

\textbf{Pre-conditions:}
\begin{enumerate*}
\item User is authenticated as Organizer
\item Current page: Manage Pupils
\end{enumerate*}

\textbf{Post-Conditions:}
\begin{enumerate*}
\item Pupil is added or updated.
\item Redirect to page: Manage Pupils
\end{enumerate*}

\textbf{Triggers:} Organizer clicks on the Edit button in the row of a Pupil inside the
table.

\end{subsubsection}
